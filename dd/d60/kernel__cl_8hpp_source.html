215 inline auto compile_kernel(const char* name,
216  const std::vector<std::string>& sources,
217  std::map<std::string, int>& options) {
218  std::string kernel_opts = "";
219  for (auto&& comp_opts : options) {
220  kernel_opts += std::string(" -D") + comp_opts.first + "="
221  + std::to_string(comp_opts.second);
222  }
223  cl::Program program(opencl_context.context(), sources);
224  try {
225  program.build({opencl_context.device()}, kernel_opts.c_str());
226 
227  return cl::Kernel(program, name);
228  } catch (const cl::Error& e) {
229  // in case of CL_BUILD_PROGRAM_FAILURE, print the build error
230  if (e.err() == -11) {
231  std::string buildlog = program.getBuildInfo<CL_PROGRAM_BUILD_LOG>(
232  opencl_context.device()[0]);
233  system_error("compile_kernel", name, e.err(), buildlog.c_str());
234  } else {
235  check_opencl_error(name, e);
236  }
237  }
238  return cl::Kernel(); // never reached because check_opencl_error throws
239 }

260  const std::map<std::string, int>& options) {
261  auto base_opts = opencl_context.base_opts();
