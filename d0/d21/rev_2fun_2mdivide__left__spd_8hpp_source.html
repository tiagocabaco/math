149  static const char *function = "mdivide_left_spd";
150  check_multiplicable(function, "A", A, "b", b);
151  check_symmetric(function, "A", A);
152  check_not_nan(function, "A", A);
153  if (A.size() == 0) {
154  return {0, b.cols()};
155  }
156 
157  // NOTE: this is not a memory leak, this vari is used in the

159  // for the returned matrix. Memory will be cleaned up with the
160  // arena allocator.

163 
164  Eigen::Matrix<var, R1, C2> res(b.rows(), b.cols());
