116  double t) const {
117  using std::vector;
118 
119  // Run nested autodiff in this scope
120  nested_rev_autodiff nested;
121 
122  vector<var> y_vars(z.begin(), z.begin() + N_);
123 
124  vector<var> dy_dt_vars = f_(t, y_vars, theta_nochain_, x_, x_int_, msgs_);
125 
126  check_size_match("coupled_ode_system", "dz_dt", dy_dt_vars.size(), "states",
127  N_);
128 
129  for (size_t i = 0; i < N_; i++) {
130  dz_dt[i] = dy_dt_vars[i].val();
131  dy_dt_vars[i].grad();
132 
133  for (size_t j = 0; j < M_; j++) {
134  // orders derivatives by equation (i.e. if there are 2 eqns
135  // (y1, y2) and 2 parameters (a, b), dy_dt will be ordered as:
136  // dy1_dt, dy2_dt, dy1_da, dy2_da, dy1_db, dy2_db
137  double temp_deriv = theta_nochain_[j].adj();
138  const size_t offset = N_ + N_ * j;
139  for (size_t k = 0; k < N_; k++) {
140  temp_deriv += z[offset + k] * y_vars[k].adj();
141  }
142 
143  dz_dt[offset + i] = temp_deriv;
144  }
145 
146  nested.set_zero_all_adjoints();
147  // Parameters stored on the outer (non-nested) nochain stack are not
148  // reset to zero by the last call. This is done as a separate step here.
149  // See efficiency note above on template specialization for more details
150  // on this.
151  for (size_t j = 0; j < M_; ++j) {
152  theta_nochain_[j].vi_->set_zero_adjoint();
153  }
154  }
