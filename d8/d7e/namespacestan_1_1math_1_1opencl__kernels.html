#ifndef A_batch
#define A_batch(i,j,k) A[(k) * cols * rows + (j) * rows + (i)]
#endif

#define A(i,j) A[(j) * rows + (i)]
#endif
#ifndef B

#endif
#ifndef C
#define C(i,j) C[(j) * rows + (i)]
#endif
// Transpose

#define BT(i,j) B[(j) * cols + (i)]
#endif
#ifndef AT
#define AT(i,j) A[(j) * cols + (i)]
#endif

#ifndef src
#define src(i,j) src[(j) * src_rows + (i)]
#endif
#ifndef dst

#endif
