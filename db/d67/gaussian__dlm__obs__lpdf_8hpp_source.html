112  if (include_summand<propto, T_y, T_F, T_G, T_V, T_W, T_m0, T_C0>::value) {
113  Eigen::Matrix<T_lp, Eigen::Dynamic, 1> m{m0};
114  Eigen::Matrix<T_lp, Eigen::Dynamic, Eigen::Dynamic> C{C0};
115  Eigen::Matrix<return_type_t<T_y>, Eigen::Dynamic, 1> yi(r);
116  Eigen::Matrix<T_lp, Eigen::Dynamic, 1> a(n);
117  Eigen::Matrix<T_lp, Eigen::Dynamic, Eigen::Dynamic> R(n, n);
118  Eigen::Matrix<T_lp, Eigen::Dynamic, 1> f(r);
119  Eigen::Matrix<T_lp, Eigen::Dynamic, Eigen::Dynamic> Q(r, r);
120  Eigen::Matrix<T_lp, Eigen::Dynamic, Eigen::Dynamic> Q_inv(r, r);
121  Eigen::Matrix<T_lp, Eigen::Dynamic, 1> e(r);

123 
124  for (int i = 0; i < y.cols(); i++) {
125  yi = y.col(i);
126  // // Predict state
127  // a_t = G_t m_{t-1}
128  a = multiply(G, m);
129  // R_t = G_t C_{t-1} G_t' + W_t
130  R = add(quad_form_sym(C, transpose(G)), W);
131  // // predict observation
132  // f_t = F_t' a_t
133  f = multiply(transpose(F), a);
134  // Q_t = F'_t R_t F_t + V_t
135  Q = add(quad_form_sym(R, F), V);
136  Q_inv = inverse_spd(Q);
137  // // filtered state
138  // e_t = y_t - f_t
139  e = subtract(yi, f);
140  // A_t = R_t F_t Q^{-1}_t
141  A = multiply(multiply(R, F), Q_inv);
142  // m_t = a_t + A_t e_t
143  m = add(a, multiply(A, e));
144  // C = R_t - A_t Q_t A_t'
145  C = subtract(R, quad_form_sym(Q, transpose(A)));
146  lp -= 0.5 * (log_determinant_spd(Q) + trace_quad_form(Q_inv, e));
147  }
148  }
149  return lp;
150 }
